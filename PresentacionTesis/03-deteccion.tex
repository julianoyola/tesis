\subsection{Detección de bordes}
\begin{frame}
	\frametitle{Detección de bordes}
	\begin{itemize}
		\item Resulta sencillo para el ser humano
		\item Borde: frontera entre el objeto y el fondo
		\item Existen varios métodos (Canny, Roberts, Sobel o Prewitt)
		\item Objetivos de un detector de borde:
		\begin{itemize}
			\item Baja tasa de error
			\item Buena localización del borde
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Método de Canny}
	\begin{itemize}
		\item Robusto contra el ruido
		\item Gran adaptabilidad
		\item Etapas del método:
		\begin{itemize}
			\item Suavizado de la imagen: Filtro gaussiano\\
			\item Obtención del gradiente: Filtro pasa altos en dirección vertical y horizontal \\
			\item Supresión de puntos que no son máximos locales: Adelgazamiento del ancho de los bordes hasta lograr bordes de un píxel de ancho\\
			\item Umbral con histéresis: Función de histéresis basada en dos umbrales; con este proceso se trata de reducir la posibilidad de aparición de contornos falsos.\\
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Operaciones morfológicas}
	\begin{itemize}
		\item Herramientas muy utilizadas en el procesamiento de imágenes
		\item Simplificar los datos de una imagen
		\item Preservar las características esenciales
		\item Eliminar aspectos irrelevantes
		\item bridge: Une pixeles que están separados
		\item Otras operaciones: open, close, clean
	\end{itemize}
\end{frame}
