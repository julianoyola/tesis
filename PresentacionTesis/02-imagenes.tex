		\item Escala

\begin{frame} 
	\frametitle{Técnicas utilizadas y medidas adoptadas}
	Técnicas y medidas adoptadas
	\begin{itemize}
		\item Elección de un subconjunto de las imágenes \pause
		\item Ecualización del histograma (Contrast-limited adaptive histogram equalization) \pause
		\item Reducción del ruido o suavización utilizando un filtro gaussiano \pause
		\item Elección del espacio de color 
	\end{itemize}
\end{frame}
