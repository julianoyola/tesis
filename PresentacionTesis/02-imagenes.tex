\begin{frame}
	\frametitle{Procesamiento digital de imágenes}
	Este problema se enmarca en el procesamiento digital de imágenes. \pause
	Algunas dificultades: \pause
	\begin{itemize}
		\item Escala \pause
		\item Elementos que afectan la imagen \pause
		\item Espacio de color \pause
		\item Detección de bordes \pause
		\item Detección de círculos \pause
		\item Falsos positivos y falsos negativos \pause
		\item Detección de la piel
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Elementos que afectan la imagen}
	Problemas
	\begin{itemize}
		\item Ruido
		\item Imperfecciones de la piel
		\item Luces y sombras
		\item Elementos ajenos
	\end{itemize}
	Técnicas
	\begin{itemize}
		\item Ecualización del histograma (Contrast-limited adaptive histogram equalization)
		\item Reducción del ruido o suavización utilizando un filtro gaussiano
	\end{itemize}
\end{frame}
